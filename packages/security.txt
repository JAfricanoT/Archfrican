# Security backbone (binary-repo only, so preflight_pkgs validates it).
# Installed by modules/60-security.sh. Opt-in tools (usbguard/apparmor/firejail)
# are NOT listed here — they're pulled on demand only when their env gate is set,
# so preflight_pkgs never forces them and they add no default surface.
nftables               # the firewall (deny-inbound / allow-outbound, named table)
pam-u2f                # FIDO2/U2F PAM module (need >= 1.3.1 — asserted in lib/fido2.sh)
libfido2               # FIDO2 device library (also enables ssh ed25519-sk keys)
bubblewrap             # unprivileged sandbox primitive (install only; opt-in use)
openssh                # ssh client + server (sshd stays OFF unless the SSH toggle is set)
